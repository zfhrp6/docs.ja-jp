 <span data-ttu-id="ba1a5-107">次の例では CS0431 が生成されます。</span><span class="sxs-lookup"><span data-stu-id="ba1a5-107">The following sample generates CS0431:</span></span>  
  
```csharp  
// CS0431.cs  
using A = Outer;  
  
public class Outer   
{  
   public class Inner   
   {  
      public static void Meth() {}  
   }  
}  
  
public class MyClass  
{  
   public static void Main()  
   {  
      A::Inner.Meth();   // CS0431  
      A.Inner.Meth();   // OK  
   }  
}  
```
